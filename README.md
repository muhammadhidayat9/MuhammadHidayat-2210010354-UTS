# Aplikasi catatan Harian
Aplikasi catatan harian adalah perangkat lunak yang dirancang untuk membantu pengguna mencatat, menyimpan, dan mengelola informasi pribadi secara digital. Aplikasi ini berfungsi seperti buku harian tradisional, namun dengan kelebihan tambahan seperti aksesibilitas, fitur pencarian, pengorganisasian berdasarkan kategori atau tanggal, dan integrasi dengan perangkat lain.

## Komponen 
- GUI : JFrame, JPanel, Jlabel, JTextField, JTable, JButton, JscrollPane
- Java class
- iText PDF : Mengekspor ke PDF
- Mysql : menyimpan data
- MVC : controller struktur data

## Fitur
- Menambahkan
- Mengubah
- Menghapus
- Menampilkan
- Pencarian
- Refresh dan
- Mengekspor

            public void mouseClicked(java.awt.event.MouseEvent evt) {
                try {
                    int selectedRow = tblCatatan.getSelectedRow(); // Ambil baris yang diklik
                    if (selectedRow != -1) {
                        DefaultTableModel model = (DefaultTableModel) tblCatatan.getModel();

                        // Ambil data dari JTable dan masukkan ke field
                        String judul = model.getValueAt(selectedRow, 1).toString(); // Kolom 1: Judul
                        String tanggalDatabase = model.getValueAt(selectedRow, 2).toString(); // Kolom 2: Tanggal (format database yyyy-MM-dd)
                        String isi = model.getValueAt(selectedRow, 3).toString(); // Kolom 3: Isi

                        // Konversi tanggal dari format database (yyyy-MM-dd) ke format input pengguna (dd/MM/yyyy)
                        SimpleDateFormat dbFormat = new SimpleDateFormat("yyyy-MM-dd");
                        SimpleDateFormat displayFormat = new SimpleDateFormat("dd/MM/yyyy");
                        Date tanggalParsed = dbFormat.parse(tanggalDatabase); // Parsing tanggal database
                        String tanggalDisplay = displayFormat.format(tanggalParsed); // Format ulang

                        // Set field input dengan data dari JTable
                        txtJudul.setText(judul);
                        txtTanggal.setText(tanggalDisplay); // Tanggal dalam format dd/MM/yyyy
                        txtIsi.setText(isi);
                    }
                } catch (ParseException ex) {
                    JOptionPane.showMessageDialog(null, "Error mengambil data: " + ex.getMessage());
                }
            }
        });

    }

    // Method untuk memuat data ke JTable
    private void loadDataToTable() {
        try {
            List<Catatan> catatanList = controller.getAllCatatan();
            DefaultTableModel model = (DefaultTableModel) tblCatatan.getModel();
            model.setRowCount(0);
            catatanList.forEach((catatan) -> {
                model.addRow(new Object[]{
                    catatan.getId(),
                    catatan.getJudul(),
                    catatan.getTanggal(),
                    catatan.getIsi()
                });
            });
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Error memuat data: " + e.getMessage());
        }
    }

        txtJudul.setText("");
        txtTanggal.setText("");
        txtIsi.setText("");
        txtJudul.requestFocus(); // Fokuskan kembali ke field pertama
    }

    private void btnTambahActionPerformed(java.awt.event.ActionEvent evt) {                                          
        try {
            String judul = txtJudul.getText().trim();
            String tanggalInput = txtTanggal.getText().trim();
            String isi = txtIsi.getText().trim();

            // Validasi input
            if (judul.isEmpty() || tanggalInput.isEmpty() || isi.isEmpty()) {
                JOptionPane.showMessageDialog(this, "Semua field harus diisi!");
                return;
            }

            // Konversi tanggal ke format database
            SimpleDateFormat inputFormat = new SimpleDateFormat("dd/MM/yyyy");
            SimpleDateFormat dbFormat = new SimpleDateFormat("yyyy-MM-dd");
            String tanggalDB = dbFormat.format(inputFormat.parse(tanggalInput));

            // Simpan data ke database
            Catatan catatan = new Catatan(0, judul, tanggalDB, isi);
            controller.createCatatan(catatan);

            // Tampilkan pesan sukses
            JOptionPane.showMessageDialog(this, "Catatan berhasil ditambahkan!");

            // Kosongkan semua field input
            resetFields();

            // Refresh data di JTable
            loadDataToTable();
        } catch (HeadlessException | SQLException | ParseException ex) {
            JOptionPane.showMessageDialog(this, "Error: " + ex.getMessage());
        }
    }                                         

    private void btnHapusActionPerformed(java.awt.event.ActionEvent evt) {                                         
        try {
            int selectedRow = tblCatatan.getSelectedRow();
            if (selectedRow == -1) {
                JOptionPane.showMessageDialog(this, "Pilih data yang akan dihapus!");
                return;
            }
            int id = (int) tblCatatan.getValueAt(selectedRow, 0);
            controller.deleteCatatan(id);
            JOptionPane.showMessageDialog(this, "Catatan berhasil dihapus!");
            loadDataToTable();
        } catch (HeadlessException | SQLException e) {
            JOptionPane.showMessageDialog(this, "Error: " + e.getMessage());
        }
    }                                        

    private void btnCariActionPerformed(java.awt.event.ActionEvent evt) {                                        
        try {
            String keyword = txtCari.getText().trim();
            if (keyword.isEmpty()) {
                JOptionPane.showMessageDialog(this, "Masukkan kata kunci untuk pencarian!");
                return;
            }
            List<Catatan> catatanList = controller.searchCatatan(keyword);
            DefaultTableModel model = (DefaultTableModel) tblCatatan.getModel();
            model.setRowCount(0);
            catatanList.forEach((catatan) -> {
                model.addRow(new Object[]{
                    catatan.getId(),
                    catatan.getJudul(),
                    catatan.getTanggal(),
                    catatan.getIsi()
                });
            });
        } catch (HeadlessException | SQLException e) {
            JOptionPane.showMessageDialog(this, "Error: " + e.getMessage());
        }
    }

    private void btnUbahActionPerformed(java.awt.event.ActionEvent evt) {                                        
        try {
            // Ambil baris yang dipilih di JTable
            int selectedRow = tblCatatan.getSelectedRow();
            if (selectedRow == -1) {
                JOptionPane.showMessageDialog(this, "Pilih data yang akan diubah!");
                return;
            }

            // Ambil data dari JTable dan field input
            int id = Integer.parseInt(tblCatatan.getValueAt(selectedRow, 0).toString()); // Kolom 0: ID
            String judul = txtJudul.getText().trim(); // Field Judul
            String isi = txtIsi.getText().trim(); // Field Isi
            String tanggalInput = txtTanggal.getText().trim(); // Field Tanggal

            // Validasi input
            if (judul.isEmpty() || tanggalInput.isEmpty() || isi.isEmpty()) {
                JOptionPane.showMessageDialog(this, "Semua field harus diisi!");
                return;
            }

            // Konversi tanggal ke format database (yyyy-MM-dd)
            SimpleDateFormat inputFormat = new SimpleDateFormat("dd/MM/yyyy");
            SimpleDateFormat dbFormat = new SimpleDateFormat("yyyy-MM-dd");
            Date tanggalParsed = inputFormat.parse(tanggalInput);
            String tanggalDB = dbFormat.format(tanggalParsed);

            // Simpan ke database
            Catatan catatan = new Catatan(id, judul, tanggalDB, isi);
            controller.updateCatatan(catatan);

            // Beri notifikasi sukses
            JOptionPane.showMessageDialog(this, "Data berhasil diubah!");

            // Refresh data di JTable
            loadDataToTable();

        } catch (HeadlessException | NumberFormatException | SQLException | ParseException ex) {
            JOptionPane.showMessageDialog(this, "Error: " + ex.getMessage());
        }

    }                                       

    private void btnExportPdfActionPerformed(java.awt.event.ActionEvent evt) {                                             
        try {
            // Pilih lokasi penyimpanan file
            JFileChooser fileChooser = new JFileChooser();
            fileChooser.setDialogTitle("Simpan sebagai");
            int userSelection = fileChooser.showSaveDialog(this);

            if (userSelection == JFileChooser.APPROVE_OPTION) {
                String filePath = fileChooser.getSelectedFile().getAbsolutePath() + ".pdf";

                // Buat dokumen PDF
                Document document = new Document();
                PdfWriter.getInstance(document, new FileOutputStream(filePath));
                document.open();

                // Tambahkan judul
                Font titleFont = FontFactory.getFont(FontFactory.HELVETICA_BOLD, 18);
                Paragraph title = new Paragraph("Data Catatan Harian", titleFont);
                title.setAlignment(Element.ALIGN_CENTER);
                document.add(title);
                document.add(new Paragraph("\n")); // Spasi

                // Tambahkan tabel
                PdfPTable table = new PdfPTable(tblCatatan.getColumnCount());
                table.setWidthPercentage(100);

                // Header Tabel
                for (int i = 0; i < tblCatatan.getColumnCount(); i++) {
                    PdfPCell cell = new PdfPCell(new Phrase(tblCatatan.getColumnName(i)));
                    cell.setHorizontalAlignment(Element.ALIGN_CENTER);
                    table.addCell(cell);
                }

                // Data Tabel
                for (int row = 0; row < tblCatatan.getRowCount(); row++) {
                    for (int col = 0; col < tblCatatan.getColumnCount(); col++) {
                        Object value = tblCatatan.getValueAt(row, col);
                        table.addCell(value != null ? value.toString() : "");
                    }
                }

                // Tambahkan tabel ke dokumen
                document.add(table);

                // Tutup dokumen
                document.close();

                // Beri notifikasi sukses
                JOptionPane.showMessageDialog(this, "Data berhasil diekspor ke PDF!");
            }
        } catch (DocumentException | HeadlessException | FileNotFoundException ex) {
            JOptionPane.showMessageDialog(this, "Error: " + ex.getMessage());
        }
    }
